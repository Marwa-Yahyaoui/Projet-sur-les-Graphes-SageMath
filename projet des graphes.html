#definition d'une fonction distance qui calcule la distance euclidienne entre deux villes
def distance( tour_Tunisie,i,j): 
    dx = tour_Tunisie[i][1]-tour_Tunisie[j][1] 
    dy = tour_Tunisie[i][2]-tour_Tunisie[j][2] 
    return(dx**2+dy**2)**0.5

#definition d'une fonction longueur tour qui retourne la longueur dun circuit
def longueur_tour(tour_Tunisie):
    d = 0 
    for i in np.arange (0 ,len (tour_Tunisie) -1) : 
        d += distance(tour_Tunisie, i, i+1) 
    d += distance(tour_Tunisie, 0, -1) 
    return(d)

#La premiere idee pour construire un chemin plus court est de partir du chemin initial. On  echange deux villes aleatoirement puis on calcule la distance du nouveau chemin. Si elle est plus courte, on conserve la modication. Si elle est plus longue, on annule cette modication. On continue tant quil nest plus possible dameliorer la distance.
def permutation(tour_Tunisie): 
    best = longueur_tour(tour_Tunisie) 
    fix = 0 
    while True: 
        i = random.randint(0, len(tour_Tunisie)-1) 
        j = random.randint(0, len(tour_Tunisie)-1) 
        if i==j: continue 
        e=tour_Tunisie[i]
        tour_Tunisie[i] = tour_Tunisie[j] 
        tour_Tunisie[j] = e 
        d = longueur_tour(tour_Tunisie) 
        if d>= best: 
            fix += 1 
            e = tour_Tunisie[i] 
            tour_Tunisie[i] = tour_Tunisie[j] 
            tour_Tunisie[j] = e 
        else: 
            best = d 
            fix = 0 
        if fix > 10000: break
    return(longueur_tour(tour_Tunisie) )

#Le resultat nest pas parfait. Parfois, les chemins se croisent, pour cela, on va essayer de retourner une partie du chemin. Il sagit de construire une fonction retourne qui retourne un chemin entre deux ville si et j. Avant lexecution, on a v  v  . . .  v  v  v  . . .  v  v  . . . i1 j1 1 2 i i+1 j apres execution, le chemin devient : v  v  . . .  v  v  v  . . .  v  v  . . .
def retourne(tour_Tunisie, i, j):
    while i <= j: 
        e = tour_Tunisie[i] 
        tour_Tunisie[i] = tour_Tunisie[j] 
        tour_Tunisie[j] = e 
        i += 1 
        j -= 1
